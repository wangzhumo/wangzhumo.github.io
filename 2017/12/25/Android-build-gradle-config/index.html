apply plugin: 'com.android.application'

android {

    //开启dataBinding
    dataBinding{
        enabled = true
    }

    defaultConfig {
        //开启multiDex
        multiDexEnabled true

        //使用jni
        ndk {
            //设置支持的SO库架构,这里的意思是只支持'armeabi', 'armeabi-v7a', 'x86'
            //这三种架构
            abiFilters 'armeabi', 'armeabi-v7a', 'x86'
        }
    }


    //设置一个编译的配置
    signingConfigs {
        //名为release的配置项
        release {
            //签名文件的位置,当前位置是根目录
            storeFile file("/keyfile.jks")
            //storePassword
            storePassword "xxxxxxxx"
            //keyAlias
            keyAlias "wangzhumo"
            //keyPassword
            keyPassword "xxxxxxxx"
            //是否使用v2Signing
            v2SigningEnabled false
        }
    }


    //编译的版本
    buildTypes {
        //debug时的配置
        debug {
            minifyEnabled false
            //这里使用上面配置的签名,可以做到debug时,也使用的是正式签名,而不是debug的 keystore
            signingConfig signingConfigs.release
            //buildConfig 的一个参数,是否开启log
            buildConfigField "boolean", "LOG_DEBUG", "true"
            proguardFiles getDefaultProguardFile('proguard-android.txt'), 'proguard-rules.pro'

        }
        release {
            // 显示Log
            minifyEnabled false
            signingConfig signingConfigs.release
            buildConfigField "boolean", "LOG_DEBUG", "false"
            proguardFiles getDefaultProguardFile('proguard-android.txt'), 'proguard-rules.pro'
        }

    }

    //adb的配置项
    adbOptions {
        //超时时间
        timeOutInMs = 5 * 1000
        //adb install -r /path..apk
        installOptions '-r'
    }

    flavorDimensions 'env'

    productFlavors {
        //正式版本
        envydemo{
            dimension 'env'
            //此处是versionName
            versionName "V1.1"
            //manifestPlaceholders
            //是manifest已经定义的参数,这里给该参数赋值
            manifestPlaceholders = [UMENG_CHANNEL_VALUE: "wangzhumoo"]
            //buildConfig 的参数,App当前的HOST地址
            buildConfigField "String","SERVER_HOST",'"http://app....com/"'
        }
        envtest {
            dimension 'env'
            versionName "Test"
            manifestPlaceholders = [UMENG_CHANNEL_VALUE: "wangzhumoo"]
            buildConfigField "String","SERVER_HOST",'"http://app.test....com/"'
        }
    }


    //自定义apk名字
    android.applicationVariants.all { variant ->
        variant.outputs.all {
            //apk的名字
            outputFileName = "App_${variant.versionName}.apk"
        }
    }

    android {
         //排除
        configurations.all {
            resolutionStrategy.force 'com.google.code.findbugs:jsr305:1.3.9'
        }
    }

    //使用java8 
    compileOptions {
        sourceCompatibility JavaVersion.VERSION_1_8
        targetCompatibility JavaVersion.VERSION_1_8
    }
}
